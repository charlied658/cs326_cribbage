package us.daveread.edu.graphics.test;

/**
 * The zones of a paddle. These affect the ball angle when hit.
 * 
 * <p>
 * Here is a link to the <a href="doc-files/PaddleLocation.html">Source
 * Code</a>.
 * </p>

 * @author readda
 */
public enum PaddleLocation {
  /**
   * The top of the paddle
   */
  TOP,
  /**
   * The area between the top and the middle of the paddle
   */
  UPPER,
  /**
   * The middle area of the paddle
   */
  MIDDLE,
  /**
   * The area between the middle and the bottom of the paddle
   */
  LOWER,
  /**
   * The bottom of the paddle
   */
  BOTTOM;
}
