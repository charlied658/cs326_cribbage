package us.daveread.edu.graphics.test;

import java.awt.Color;
import java.util.Random;

import us.daveread.edu.graphics.surface.DrawingSurface;
import us.daveread.edu.graphics.surface.MainFrame;
import us.daveread.edu.graphics.tool.Turtle;
import us.daveread.edu.utilities.Utility;

/**
 * <a href="doc-files/TurtleTest.html"><img src="doc-files/TurtleTest.png" alt=
 * "Screenshot" style="float:right;width:100px;height:auto;border:1px solid
 * #000000"></a>
 * Various test of the Turtle language support library. To use, create an
 * instance and run it on a thread.
 * 
 * <pre>
 * 
 * public static void main(String[] args) {
 *   new Thread(new TurtleTest()).start();
 * }
 * </pre>
 * 
 * <p>
 * Here is a link to the <a href="doc-files/TurtleTest.html">Source Code</a>.
 * </p>
 * 
 * @author readda
 */
public class TurtleTest extends DrawingSurface implements Runnable {
  /**
   * The serial version id
   */
  private static final long serialVersionUID = 1L;

  public TurtleTest() {
    new MainFrame(this, "Turtle Test", 500, 400);
  }

  public void run() {
    resetAndClearTest();
    square1();
    dropTurtle();
    triangle1();
    writeNumbers();
    writeWords();
    path();
    square2();
    spiral1();
    circle1();
    spiral2();
    crosshair();
    multipleTurtles();
  }

  public void multipleTurtles() {
    Turtle ann = new Turtle(this, "ann");
    Turtle bob = new Turtle(this, "bob");
    Turtle carly = new Turtle(this, "carly");
    Turtle noname = new Turtle(this);

    ann.forward(150);
    bob.right(90);
    bob.forward(150);
    carly.right(180);
    carly.forward(150);
    noname.left(90);
    noname.forward(150);

    noname.hideTurtle();
    bob.hideTurtle();

    ann.left(160);
    ann.forward(100);

    noname.right(90);
    noname.forward(50);

    noname.showTurtle();
    bob.showTurtle();
  }

  public void resetAndClearTest() {
    Turtle turtle = new Turtle(this);
    turtle.hideTurtle();
    turtle.forward(100);
    turtle.reset(); // Should re-center and re-show turtle
    turtle.forward(100);
    Utility.pause(2000);
    turtle.clear();
    turtle.forward(100);
    Utility.pause(2000);
    turtle.reset();
    turtle.hideTurtle();
  }

  public void square1() {
    Turtle turtle = new Turtle(this);
    turtle.forward(100);
    turtle.right(90);
    turtle.forward(100);
    turtle.right(90);
    turtle.forward(100);
    turtle.right(90);
    turtle.forward(100);
    turtle.hideTurtle();

  }

  public void triangle1() {
    Turtle turtle = new Turtle(this);
    turtle.penUp();
    turtle.setPosition(100, 100);
    turtle.penDown();
    turtle.color(Color.red);
    turtle.forward(100);
    turtle.right(120);
    turtle.forward(100);
    turtle.right(120);
    turtle.forward(100);
    turtle.hideTurtle();
  }

  public void path() {
    Turtle turtle = new Turtle(this);
    turtle.speed(1);
    turtle.penUp();
    turtle.setPosition(0, 0);
    turtle.penDown();
    turtle.color(250, 180, 64);
    turtle.setHeading(45);
    turtle.forward(75);
    turtle.setHeading(90);
    turtle.forward(75);
    turtle.setHeading(105);
    turtle.forward(75);
    turtle.setHeading(20);
    turtle.forward(75);
    turtle.setHeading(270);
    turtle.forward(75);
    turtle.hideTurtle();
  }

  public void writeNumbers() {
    Turtle turtle = new Turtle(this);
    turtle.penUp();
    turtle.setPosition(30, 180);
    turtle.penUp();
    for (int i = 0; i < 8; ++i) {
      turtle.write(i);
      turtle.forward(20);
    }
    turtle.hideTurtle();
  }

  public void writeWords() {
    Turtle turtle = new Turtle(this);
    turtle.penUp();
    turtle.setPosition(30, 380);
    turtle.penDown();
    turtle.write("Hello World");
    turtle.hideTurtle();
  }

  public void square2() {
    Turtle turtle = new Turtle(this);
    turtle.color(Color.green);
    turtle.penUp();
    turtle.setPosition(210, 50);
    turtle.penDown();
    for (int i = 0; i < 4; ++i) {
      turtle.forward(100);
      turtle.right(90);
    }
    turtle.hideTurtle();
  }

  public void spiral1() {
    Turtle turtle = new Turtle(this);
    turtle.color(Color.green);
    turtle.speed(10);
    for (int i = 0; i < 30; ++i) {
      for (int j = 0; j < 4; ++j) {
        turtle.forward(100);
        turtle.right(90);
      }
      turtle.right(25);
    }
    turtle.hideTurtle();
  }

  public void circle1() {
    Turtle turtle = new Turtle(this);
    turtle.penUp();
    turtle.setPosition(60, 60);
    turtle.penDown();
    turtle.hideTurtle();
    for (int i = 0; i < 360; ++i) {
      turtle.forward(1);
      turtle.right(1);
    }
    turtle.hideTurtle();
  }

  public void spiral2() {
    Turtle turtle = new Turtle(this);
    turtle.color(Color.blue);
    turtle.speed(0);
    turtle.penUp();
    turtle.setPosition(380, 280);
    turtle.penDown();
    for (int i = 0; i < 30; ++i) {
      if (i == 25) {
        turtle.speed(9);
        turtle.showTurtle();
      }
      for (int j = 0; j < 360; ++j) {
        turtle.forward(1);
        turtle.right(1);
      }
      turtle.right(25);
    }
    turtle.hideTurtle();
  }

  public void crosshair() {
    Turtle turtle = new Turtle(this);
    turtle.penUp();
    turtle.setPosition(60, 310);
    turtle.penDown();
    turtle.speed(9);
    for (int i = 0; i < 18; ++i) {
      turtle.color(Color.yellow);
      turtle.penDown();
      turtle.forward(50);
      turtle.penUp();
      turtle.backward(50);
      turtle.color(Color.green);
      turtle.penDown();
      turtle.backward(50);
      turtle.penUp();
      turtle.forward(50);
      turtle.color(Color.red);
      turtle.penDown();
      turtle.right(90);
      turtle.forward(50);
      turtle.penUp();
      turtle.backward(50);
      turtle.color(Color.blue);
      turtle.penDown();
      turtle.backward(50);
      turtle.penUp();
      turtle.forward(50);
      turtle.left(95);
    }
    turtle.hideTurtle();
  }

  public void dropTurtle() {
    Turtle turtle = new Turtle(this);
    Random rand = new Random();
    for (int i = 0; i < 10; i++) {
      turtle.setPosition(rand.nextInt(400), rand.nextInt(400));
      Utility.pause(500);
    }
    turtle.hideTurtle();
  }

  public static void main(String[] args) {
    new Thread(new TurtleTest()).start();
  }
}
